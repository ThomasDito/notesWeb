state = data kita yang bisa berubah-ubah nilainya, kemudian data ini disimpan dalam function yang bernama useState

useState = function milik react library yang dapat merubah atau memainkan state ini

useState = state + setState

setState = function untuk mengubah nilai state

cara import -> import {useState} from 'react'
syntax -> 'const[state, setState] = useState(initialState)'

initialState = nilai awal dari state

conditional rendering = function boolean yang akan render konten jika nilainya true, dan jika false maka tidak akan ditambilkan

syntax -> {namaVariableBoolean && (isiKonten)}
